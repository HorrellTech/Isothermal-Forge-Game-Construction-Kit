var o = object_add();

o.draw = function()
{
    for(var i = 0; i < 360; i += 360 / 10)
    {
        var d = i + delta_time * 3;
        var dx = lengthdir_x(64, d);
        var dy = lengthdir_y(32, d);
        var r = 10 + dy / 10;
        draw_set_color(rgb(r * 10, r * 10, r * 10));
        draw_circle(this.x + dx, this.y + dy, r, false);
        draw_set_color(c_red);
        draw_circle(this.x, this.y - 16 + pulse(10, 6), 16, false);
    }
}

instance_create(room_width / 2, room_height / 2, o);
