var o = object_add();

o.awake = function()
{
    this.grid = [];//create_array(840, 240);

    for(var i = 0; i < 840; i += 1)
    {
        for(var j = 0; j < 240; j += 1)
        {
            this.grid.push([i, j]);
            //this.grid[i, j] = irandom_range(-1, 4);
            this.grid[i][j] = 0; //irandom_range(-1, 4);
        }
    }

    // Generate
    var xx, yy;
    yy = 4;
          
    for(xx = 0; xx < 840; xx += 1)
    {
        this.grid[xx][yy] = 1;        

        for(var i = yy + 1; i < 240 - yy; i += 1)
        {
            this.grid[xx][i] = irandom_range(2, 4);
        }

        yy += irandom_range(-1, 2);
    }
}

o.draw = function()
{
    view_xview += 2;
    //view_yview += 2;
    for(var i = floor(view_xview / 32); i < ceil((view_xview + view_wview) / 32); i += 1)
    {
        for(var j = floor(view_yview / 32); j < ceil((view_yview + view_hview) / 32); j += 1)
        {
            switch(this.grid[i][j])
            {
                case 1:
                    draw_set_color(c_blue);
                    break;                
                case 2:
                    draw_set_color(c_green);
                    break;
                case 3:
                    draw_set_color(c_lime);
                    break;
            }

            if(this.grid[i][j] > 0)
            {
                draw_rectangle(i * 32, j * 32, (i * 32) + 32, (j * 32) + 32, false);
            }
            draw_set_color(c_white);
        }
    }
}

instance_create(0, 0, o);
